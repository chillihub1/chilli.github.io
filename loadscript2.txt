-- Gradient animation (giữ nguyên)
do
    local tweenService = game:GetService("TweenService")
    local grad = Instance.new("UIGradient", dpanel)
    grad.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(64,68,75)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(54,57,63)),
    }
    spawn(function()
        while dpanel.Parent do
            tweenService:Create(grad, TweenInfo.new(20), {Rotation = 360}):Play()
            wait(20)
            grad.Rotation = 0
        end
    end)
end
